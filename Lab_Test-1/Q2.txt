class Queue:
    def __init__(self):
        """Initializes an empty Queue."""
        self._items = []

    def enqueue(self, value):
        """Adds an element to the rear of the queue.

        Args:
            value: The element to add to the queue.
        """
        self._items.append(value)

    def dequeue(self):
        """Removes and returns the front element from the queue.

        Returns:
            The front element of the queue.

        Raises:
            IndexError: If the queue is empty.
        """
        if self.is_empty():
            raise IndexError("dequeue from an empty queue")
        return self._items.pop(0)

    def peek(self):
        """Returns the front element of the queue without removing it.

        Returns:
            The front element of the queue.

        Raises:
            IndexError: If the queue is empty.
        """
        if self.is_empty():
            raise IndexError("peek from an empty queue")
        return self._items[0]

    def is_empty(self):
        """Checks if the queue is empty.

        Returns:
            True if the queue is empty, False otherwise.
        """
        return len(self._items) == 0

    def size(self):
        """Returns the number of elements in the queue.

        Returns:
            The number of elements in the queue.
        """
        return len(self._items)

    def __str__(self):
        """Returns a string representation of the queue."""
        return f"Queue({list(self._items)})"

# Example Usage:
print("--- Initializing Queue ---")
my_queue = Queue()
print(f"Is queue empty? {my_queue.is_empty()}")
print(f"Queue size: {my_queue.size()}")
print(f"Queue content: {my_queue}")

print("\n--- Enqueueing elements ---")
my_queue.enqueue(10)
my_queue.enqueue('hello')
my_queue.enqueue(True)
print(f"Is queue empty? {my_queue.is_empty()}")
print(f"Queue size: {my_queue.size()}")
print(f"Queue content: {my_queue}")

print("\n--- Peeking at the front element ---")
print(f"Front element: {my_queue.peek()}")
print(f"Queue content after peek: {my_queue}")

print("\n--- Dequeueing elements ---")
dequeued_item = my_queue.dequeue()
print(f"Dequeued: {dequeued_item}")
print(f"Queue size: {my_queue.size()}")
print(f"Queue content: {my_queue}")

dequeued_item = my_queue.dequeue()
print(f"Dequeued: {dequeued_item}")
print(f"Queue size: {my_queue.size()}")
print(f"Queue content: {my_queue}")

print("\n--- Enqueueing more elements ---")
my_queue.enqueue(20)
my_queue.enqueue(30)
print(f"Queue content: {my_queue}")

print("\n--- Dequeueing remaining elements ---")
while not my_queue.is_empty():
    print(f"Dequeued: {my_queue.dequeue()}")
print(f"Is queue empty? {my_queue.is_empty()}")
print(f"Queue size: {my_queue.size()}")

print("\n--- Trying to dequeue from an empty queue ---")
try:
    my_queue.dequeue()
except IndexError as e:
    print(f"Error: {e}")
